import java.util.Scanner;

public class Main{
    
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        String word = sc.next();
        String reverse = "";
        
        for (int i = word.length() - 1 ; i >= 0; i--) {
            reverse = reverse + word.charAt(i);
        }
        
        if (word.equals(reverse)) {
            System.out.print(1);
        } else {
            System.out.print(0);
        }
    }
}
